function zaregistrujObsluhuUdalosti(htmlElement, typUdalosti, obsluhaUdalosti) {
  if (htmlElement.addEventListener) {
    htmlElement.addEventListener(typUdalosti, obsluhaUdalosti, false);
  } else {
    htmlElement.attachEvent('on' + typUdalosti, obsluhaUdalosti);
  }
}
